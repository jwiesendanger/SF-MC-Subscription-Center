%%[ /* UK Parent BU — Subscription Preference Center Phase 4 | SF-MC Subscription Center POC Flow: Email footer → Child Relay (action=prefs / action=unsub) → Relay reads CRM, builds URL → Parent (action=load / action=unsub) → Subscriber submits form → POST → meta refresh to Relay (action=write) → Relay writes CRM → Relay redirects to Parent (action=confirm) Parameters received from relay page: action "load" | "unsub" | "confirm" sub MC Subscriber Key mid Child BU MID oid Salesforce Org ID (18-char) — used to look up config DE writeback Child relay page URL (relay's own URL, for write-back redirect) globalunsub "true" if subscriber is already globally unsubscribed [FieldAPIName] One param per active field, value "true" or "false" e.g. Email_Opt_In_Benefits__c=true IMPORTANT PATTERNS: - meta http-equiv="refresh" for ALL post-submit redirects (Redirect() and HTTPRedirect() cause "page content contains errors") - Relay page passes current CRM field values as query params (parent does NOT query the synced DE directly) - Field list is driven by SubscriptionConfig_Fields DE (admin tool) No field names are hardcoded here */ VAR @action, @sub, @mid, @oid, @writeback, @orgname VAR @globalunsub, @isPost, @formGlobal, @writeURL, @errorMsg VAR @configFields, @row, @i, @apiName, @label, @isActive, @fieldVal, @currentVal, @fieldCount /* ── Read GET parameters ──────────────────────────────────────────── */ SET @action = QueryParameter("action") SET @sub = QueryParameter("sub") SET @mid = QueryParameter("mid") SET @oid = QueryParameter("oid") SET @writeback = QueryParameter("writeback") SET @globalunsub = QueryParameter("globalunsub") SET @isPost = "false" /* ── POST: subscriber submitted the preference form ──────────────── */ IF RequestParameter("formAction") == "saveprefs" THEN SET @isPost = "true" SET @sub = RequestParameter("sub") SET @mid = RequestParameter("mid") SET @oid = RequestParameter("oid") SET @writeback = RequestParameter("writeback") SET @formGlobal = RequestParameter("globalunsub") ENDIF /* ── Load org display name from SubscriptionConfig_Orgs ─────────── */ IF NOT Empty(@oid) THEN SET @orgname = Lookup("SubscriptionConfig_Orgs", "DisplayName", "OrgID", @oid) ENDIF IF Empty(@orgname) THEN SET @orgname = "University of Kentucky" ENDIF /* ── Load active field config for this org ───────────────────────── */ /* Retrieves all fields registered for this org via the Phase 2 admin tool.
